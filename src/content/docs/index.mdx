import { getCollection } from "astro:content";

export const latestPosts = (await getCollection("docs"))
  .filter(({ id, data }) =>
    id !== "index" &&
    id !== "about" &&
    !id.startsWith("secret/") &&
    data.updatedAt
  )
  .sort((a, b) => b.data.updatedAt.getTime() - a.data.updatedAt.getTime())
  .slice(0, 3);

<section class="recent-posts home-page" id="recent-posts">
  <div class="home-section-heading">
    <span class="home-eyebrow">LATEST NOTES</span>
  </div>

  <div class="post-list">
    {latestPosts.map(({ id, data }, index) => (
      <a class={`post-card${index === 0 ? " post-card--featured" : ""}`} href={`/${id}/`}>
        <h3>{data.title}</h3>
        <p>{data.description}</p>
        <div class="post-card__meta">
          <time datetime={data.updatedAt.toISOString()}>
            {data.updatedAt.toLocaleDateString("ko-KR", {
              year: "numeric",
              month: "long",
              day: "numeric",
              timeZone: "UTC",
            })}
          </time>
          <strong>READ ARTICLE <span aria-hidden="true">→</span></strong>
        </div>
      </a>
    ))}
  </div>
</section>
